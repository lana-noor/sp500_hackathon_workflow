sector_map = tickers.set_index('ticker')['sector'].to_dict()

rets = returns[common]

sector_returns = {}
for sector in tickers['sector'].unique():
    cols = [t for t in common if sector_map[t] == sector]
    if cols:
        sector_returns[sector] = rets[cols].mean(axis=1)
sector_returns = pd.DataFrame(sector_returns)
